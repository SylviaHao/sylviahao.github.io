olympics$age_na <- as.numeric(is.na(olympics$age))
olympics$height_na <- as.numeric(is.na(olympics$height))
olympics$weight_na <- as.numeric(is.na(olympics$weight))

#Is missingness associated with year?
olympics_na_df <- olympics |> 
         group_by(year, sport) |> 
         summarize(height_na_prop = sum(height_na)/n(),
                   weight_na_prop = sum(weight_na)/n(),
                   age_na_prop = sum(age_na)/n()) |> 
  tidyr::pivot_longer(cols = height_na_prop:age_na_prop,
                      names_to = 'biometric',
                      values_to = 'prop')

ggplot(olympics_na_df |> filter(sport == 'Swimming'),
  aes(x = year, y = prop, color = sport)) +
    geom_point() + facet_grid(.~ biometric)

swim_100_men_df <- olympics |> filter(sport == 'Swimming' & year >= 1960) |> filter(event == 'Swimming Men\'s 100 metres Freestyle') |> select(age, height, weight, year, medal) |> mutate(medalist = dplyr::if_else(is.na(medal), true = '0', false = '1')) |> mutate(Sex = 'Men')


swim_100_somen_df <- olympics |> filter(sport == 'Swimming' & year >= 1960) |> filter(event == 'Swimming Women\'s 100 metres Freestyle') |> select(age, height, weight, year, medal) |> mutate(medalist = dplyr::if_else(is.na(medal), true = '0', false = '1')) |> mutate(Sex = 'women')
